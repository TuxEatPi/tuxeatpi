        self.move_wings_thread.start()
        try:
            super(FakeWings, self).move_start()
        except Exception:  # pylint: disable=W0703
            pass

[docs]    def run(self):
        """Start moving wings"""
        # Get pin_id from self.pins
        pin_id = GPIO.GPIO_TO_PIN[self.position_pin]
        self.running = True
        while self.running:
            if self.running:
                set_pin_value(pin_id, 1)
                time.sleep(0.1)
            if self.running:
                set_pin_value(pin_id, 0)
                time.sleep(0.1)
            if self.running:
                set_pin_value(pin_id, 1)
                time.sleep(0.1)
            if self.running:
                set_pin_value(pin_id, 0)
                time.sleep(0.25)
